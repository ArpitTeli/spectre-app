// Exports road segments as position + direction
// Check %LOCALAPPDATA%\Arma 3\Arma3_x64_*.rpt for SPECTRE_ROADS: lines

hint "SPECTRE road export starting..."; diag_log "SPECTRE_ROADS:START"; _count = 0; _step = 50; _seen = []; for [{_y = 0}, {_y < 8192}, {_y = _y + _step}] do { for [{_x = 0}, {_x < 8192}, {_x = _x + _step}] do { _roads = [_x, _y, 0] nearRoads (_step * 0.75); { _p = getPosATL _x; _rx = _p select 0; _ry = _p select 1; if (_rx > 0 && _rx < 8192 && _ry > 0 && _ry < 8192) then { _key = format ["%1_%2", round (_rx / 5), round (_ry / 5)]; if !(_key in _seen) then { _seen pushBack _key; _conn = roadsConnectedTo _x; _w = 6; if (count _conn > 0) then { _c1 = getPosATL (_conn select 0); _dx = (_c1 select 0) - _rx; _dy = (_c1 select 1) - _ry; if (_dx * _dx + _dy * _dy > 1) then { _w = 8; }; }; diag_log format ["SPECTRE_ROADS:R,%1,%2,%3,%4", round (_rx * 100) / 100, round (_ry * 100) / 100, round (getDir _x * 10) / 10, _w]; _count = _count + 1; }; }; } forEach _roads; }; if (_y mod 500 == 0) then { hint format ["Exporting roads... y=%1 count=%2", _y, _count]; }; }; diag_log "SPECTRE_ROADS:END"; hint format ["Done - %1 road segments exported. Check RPT.", _count]
